C++ & Python

mir->jsonSpewer().spewMIR(mir->graph());
def invoke(self, mirExpr, from_tty):
    mir = gdb.parse_and_eval(mirExpr)
    jsonSpewer = mir['gs_']['jsonSpewer_']
    graph = mir['graph_']
    gdb.parse_and_eval('(*(%s*)(%s)).spewMIR((%s)%s)' %
        (jsonSpewer.type, jsonSpewer.address,
         graph.type, graph,))
